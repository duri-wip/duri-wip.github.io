AirflowВЌљВёю ContextВЎђ args, kwargsВЮў ьЎюВџЕ

AirflowВЌљВёю args,kwargsвЦ╝ ВѓгВџЕьЋўвЕ┤ ьїїВЮ┤ВЇг ьЋеВѕўВЮў ВЮИВъљвЦ╝ ВюаВЌ░ьЋўЖ▓ї вІцвБ░ Вѕў ВъѕвІц. ВЮ┤ Ж░ювЁљЖ│╝ ьЋеЖ╗ў airflowВЮў contextьЎюВџЕ в░Ев▓ЋВЮё ьЋеЖ╗ў ВёцвфЁьЋювІц.

argsВЎђ kwargsвъђ?

args ВѓгВџЕв▓Ћ

*argsвіћ ВЌгвЪг Ж░юВЮў Ж░њВЮё ьіюьћї ьўЋьЃювАю в░ЏвіћвІц. ьЋеВѕўВЌљВёю ВЮИВъљВЮў Ж░юВѕўЖ░ђ в»Ивдг ВаЋьЋ┤ВаИ ВъѕВДђ ВЋіВЮё вЋї ВюаВџЕьЋўвІц. ВЮИвЇ▒ВіцвЦ╝ ВѓгВџЕьЋ┤ Ж░њВЮё Ж║╝вѓ╝ Вѕў ВъѕВДђвДї, Вќ┤вќц Ж░њВЮ┤ Вќ┤вќц ВЮИвЇ▒ВіцВЌљ ВюёВ╣ўьЋўвіћВДђ вфЁьЎЋьЋўВДђ ВЋіВЮђ вІеВаљВЮ┤ ВъѕвІц.

def regist(name, sex, *args):
    print(f'name: {name}\nsex: {sex}')
    address = args[0] if len(args) > 0 else None
    p_num = args[1] if len(args) > 1 else None
    print(f'address: {address}\nphone number: {p_num}')

# ьЋеВѕў ьўИВХю ВўѕВІю
regist('Alice', 'Female', 'Seoul', '010-1234-5678')

Вюё ВўѕВІюВЌљВёю nameЖ│╝ sexвіћ ьЋёВѕў ВЮИВъљвАю ВиеЖИЅвљўЖ│а, *argsвіћ ВХћЖ░ђВаЂВЮИ ВЮИВъљвЦ╝ в░ЏвіћвІц. name, sexвіћ ьЋёВѕўВаЂВю╝вАю в░ЏВЋёВўцвіћвЇ░ ЖиИ вњцВЮў ВЮИВъљвЊцВЮђ ВъёВЮўВаЂВю╝вАю ВА┤ВъгьЋа Вѕў Въѕвіћ Ж▓йВџ░ВЌљ ВЮИВъљЖ░ђ 2Ж░ю ВъѕВю╝вЕ┤ 2Ж░ювЦ╝ вІц в░ЏВЋёВўцЖ│а, 1Ж░ювДї ВъѕВю╝вЕ┤ 1Ж░ювДї вЇћ вДЈВЋёВўцЖ│а, ВЌєВю╝вЕ┤ ВЌєвіћвїђвАю ВиеЖИЅьЋўвіћ ВюаВЌ░ьЋю В▓ўвдгЖ░ђ Ж░ђвіЦьЋўвІц.

kwargs ВѓгВџЕв▓Ћ

**kwargsвіћ ьѓц-Ж░њ ВїЇВЮў ьўЋьЃювАю ВЌгвЪг ВЮИВъљвЦ╝ в░ЏвіћвІц. вћЋВЁћвёѕвдгВ▓ўвЪ╝ Ж░њВЮё вІцвБ░ Вѕў ВъѕВќ┤, ВЮИВъљ ВЮ┤вдёВЮё ВДђВаЋьЋ┤ Ж░њВЮё Ж║╝вѓ╝ Вѕў Въѕвіћ ВъЦВаљВЮ┤ ВъѕвІц.

def regist(name, sex, **kwargs):
    print(f'name: {name}\nsex: {sex}')
    address = kwargs.get('address', None)
    p_num = kwargs.get('p_num', None)
    print(f'address: {address}\nphone number: {p_num}')

# ьЋеВѕў ьўИВХю ВўѕВІю
regist('Alice', 'Female', address='Seoul', p_num='010-1234-5678')

kwargs.get()ВЮё ВѓгВџЕьЋўвЕ┤ ьѓцЖ░ђ ВА┤ВъгьЋўВДђ ВЋіВЮё вЋї ЖИ░в│ИЖ░њВЮё ВёцВаЋьЋа Вѕў ВъѕВіхвІѕвІц.

argsВЎђ kwargs ьЋеЖ╗ў ВѓгВџЕьЋўЖИ░

argsВЎђ kwargsвіћ ьЋеЖ╗ў ВѓгВџЕьЋа Вѕў ВъѕвІц. ВЮ┤вЋї, *argsЖ░ђ ве╝Вађ, **kwargsвіћ ЖиИ вњцВЌљ ВўцвЈёвАЮ ВѓгВџЕьЋювІц.

def sample_func(a, b, *args, **kwargs):
    print(f"a: {a}, b: {b}")
    print(f"args: {args}")
    print(f"kwargs: {kwargs}")

AirflowВЌљВёюВЮў Context ьЎюВџЕ

AirflowВЮў В╗еьЁЇВіцьіИвіћ ьі╣ВаЋ ьЃюВіцьЂгВЌљ вїђьЋю ВІцьќЅ ВаЋв│┤вЦ╝ вІ┤Ж│а Въѕвіћ вћЋВЁћвёѕвдгвІц. ВЮ┤вЦ╝ ьєхьЋ┤ ьўёВъг ВІцьќЅ ВцЉВЮИ DAG, ьЃюВіцьЂг, ВІцьќЅ ВІюЖ░ё вЊ▒ вІцВќЉьЋю ВаЋв│┤вЦ╝ ьїївЮ╝в»Иьё░вАю ВаёвІгв░ЏВЮё Вѕў ВъѕвІц. @task вЇ░ВйћваѕВЮ┤ьё░вѓў PythonOperatorвЦ╝ ВѓгВџЕьЋа вЋї, contextвЦ╝ ьЋеВѕўВЌљ ВаёвІгьЋўВЌг ВІцьќЅ ьЮљвдёВЌљ вїђьЋю ВаЋв│┤вЦ╝ ьЎюВџЕьЋа Вѕў ВъѕвІц.

ContextВЌљ ьЈгьЋевљю ЖИ░в│И ьЋГвфЕ

  • dag: ьўёВъг ВІцьќЅ ВцЉВЮИ DAG Ж░ЮВ▓┤
  • dag_run: ьўёВъг ВІцьќЅ ВцЉВЮИ DAGRun Ж░ЮВ▓┤
  • task: ьўёВъг ВІцьќЅ ВцЉВЮИ ьЃюВіцьЂг Ж░ЮВ▓┤
  • task_instance: ьЃюВіцьЂг ВЮИВіцьё┤Віц Ж░ЮВ▓┤
  • execution_date: ьЃюВіцьЂг ВІцьќЅ вѓаВДю в░Ј ВІюЖ░ё
  • ds: ВІцьќЅ вѓаВДю (YYYY-MM-DD ьўЋВІЮ)
  • ts: ВІцьќЅ ВІюЖ░ё (YYYY-MM-DDTHH:MM:SS ьўЋВІЮ)
  • prev_execution_date: ВЮ┤Ваё ВІцьќЅ вѓаВДю в░Ј ВІюЖ░ё
  • next_execution_date: вІцВЮї ВІцьќЅ вѓаВДю в░Ј ВІюЖ░ё
  • conf: Airflow ВаёВ▓┤ ВёцВаЋ ВаЋв│┤ (configuration)
  • macros: AirflowВЌљВёю ВѓгВџЕьЋа Вѕў Въѕвіћ вДцьЂгвАю ьЋеВѕўвЊц

Context ВўѕВІю: вДцьЂгвАю ьЎюВџЕ

DAG вѓ┤ВЌљВёю ьЃюВіцьЂгЖ░ђ ВІцьќЅвља вЋї, context ВаЋв│┤вЦ╝ в░ЏВЋё ьўёВъг ВІцьќЅ ВцЉВЮИ вѓаВДюВЎђ ВЮ┤Ваё, вІцВЮї ВІцьќЅ вѓаВДювЦ╝ ВХюваЦьЋЕвІѕвІц.

from airflow import DAG
from airflow.operators.python import PythonOperator
from airflow.utils.dates import days_ago
from datetime import timedelta

# ЖИ░в│И DAG ВёцВаЋ
default_args = {
    'owner': 'airflow',
    'start_date': days_ago(1),
}

# вДцьЂгвАювЦ╝ ВѓгВџЕьЋўвіћ ьЋеВѕў
def use_macros(**context):
    ds = context['ds']  # ьўёВъг ВІцьќЅ вѓаВДю (YYYY-MM-DD ьўЋВІЮ)
    next_ds = context['next_execution_date']  # вІцВЮї ВІцьќЅ вѓаВДю
    prev_ds = context['prev_execution_date']  # ВЮ┤Ваё ВІцьќЅ вѓаВДю

    print(f"ьўёВъг ВІцьќЅ вѓаВДю: {ds}")
    print(f"вІцВЮї ВІцьќЅ вѓаВДю: {next_ds}")
    print(f"ВЮ┤Ваё ВІцьќЅ вѓаВДю: {prev_ds}")

# DAG ВЃЮВё▒
with DAG(
    'macro_example_dag',
    default_args=default_args,
    schedule_interval=timedelta(days=1),  # вДцВЮ╝ ВІцьќЅ
    catchup=False,
) as dag:

    # PythonOperatorВЌљВёю вДцьЂгвАю ьЎюВџЕ
    macro_task = PythonOperator(
        task_id='use_macros_task',
        python_callable=use_macros,
        provide_context=True,  # context ВаЋв│┤вЦ╝ ьЋеВѕўВЌљ ВаёвІг
    )

use_macros ьЋеВѕўЖ░ђ context ВаЋв│┤вЦ╝ ВѓгВџЕьЋ┤ ьўёВъгВЎђ ВЮ┤Ваё, ЖиИвдгЖ│а вІцВЮї ВІцьќЅ вѓаВДювЦ╝ ВХюваЦьЋювІц. provide_context=True ВўхВЁўВЮё ВѓгВџЕьЋ┤ PythonOperatorвіћ ВъљвЈЎВю╝вАю ьЋ┤вІ╣ ВаЋв│┤вЦ╝ ьЋеВѕўвАю ВаёвІгьЋювІц.
