вЇ░ВйћваѕВЮ┤ьё░віћ ЖИ░ВА┤ ьЋеВѕўвЦ╝ Ж░љВІИВёю в░ћЖ╣ЦВЌљ ВХћЖ░ђ ЖИ░віЦВЮё вЇДвХЎВЮ┤віћ ьїїВЮ┤ВЇг вгИв▓Ћ. ьЋеВѕўВЮў ВЮИВъљвАю вўљ вІцвЦИ ьЋеВѕўвЦ╝ ВаёвІгьЋўВЌг, ьЋеВѕўВЮў ВІцьќЅ ВаёьЏёвАю ьі╣ВаЋ ВъЉВЌЁВЮё В▓ўвдгьЋўЖ▒░вѓў ьЋеВѕўвЦ╝ ВѕўВаЋьЋа Вѕў Въѕвіћ ВюаВџЕьЋю в░Ев▓ЋВЮ┤вІц.

ВўѕвЦ╝ вЊцВќ┤ ВЌгвЪгЖ░ђВДђ ьЋеВѕўЖ░ђ ВъѕвіћвЇ░, ВЮ┤ ьЋеВѕўвЊцВЮў ВІцьќЅ ВаёЖ│╝ ьЏёВЌљ вАюЖиИвЦ╝ ВХюваЦьЋ┤ВЋ╝ьЋўвіћ ВЃЂьЎЕВЮ┤ Въѕвіћ Ж▓йВџ░ВЌљ ВЮ┤ВЌљ вїђьЋю ьЋ┤Ж▓░в░Ев▓ЋВЮђ ВЮ┤вЪ░ Ж▓йВџ░вЊцвАю вѓўвѕёВќ┤ в│╝ Вѕў ВъѕвІц.

def log_decorator(func):
    def wrapper(*args, **kwargs):
        print(f'{func.__name__} ВІюВъЉ')
        result = func(*args, **kwargs)
        print(f'{func.__name__} вЂЮ')
        return result
    return wrapper

from airflow.decorators import dag, task
from airflow.utils.dates import days_ago

@dag(schedule_interval='@daily', start_date=days_ago(2), catchup=False)
def dag():

    @task
    def task_1():
        print("ьЃюВіцьЂг 1 ВІцьќЅ")

    @task
    def task_2():
        print("ьЃюВіцьЂг 2 ВІцьќЅ")

    task_1() >> task_2()

dag_instance = dag()

ьЋю ьЃюВіцьЂгВЮў ВХюваЦВЮё ВъЁваЦВю╝вАю в░ЏВЋёВёю вфЁва╣ВЮё ВѕўьќЅьЋўвіћ ьЃюВіцьЂгвЦ╝ ЖхгВё▒ьЋа Вѕў ВъѕвІц.

from airflow.decorators import dag, task
from airflow.utils.dates import days_ago

@dag(schedule_interval='@daily', start_date=days_ago(2), catchup=False)
def data_pipeline():

    @task
    def extract_data():
        return {"name": "Airflow", "type": "Pipeline"}

    @task
    def process_data(data):
        print(f"вЇ░ВЮ┤ьё░ В▓ўвдг ВцЉ: {data['name']}ВЮђ {data['type']}ВъЁвІѕвІц.")

    data = extract_data()
    process_data(data)

dag_instance = data_pipeline()
