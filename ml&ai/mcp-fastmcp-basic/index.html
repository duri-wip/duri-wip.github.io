from fastmcp import FastMCP
from typing import Dict

app = FastAPI()

)

@mcp.tool
def plus(a: int, b: int) -> int:
"""
л‘җ мҲ«мһҗлҘј лҚ”н•ҳлҠ” лҸ„кө¬мһ…лӢҲлӢӨ.

        a: мІ« лІҲм§ё мҲ«мһҗ

    Returns:
        л‘җ мҲ«мһҗмқҳ н•©

    Examples:
        - a=5, b=3 -> 8
    """
    return a + b
