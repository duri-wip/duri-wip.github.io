1вХђВЌљВёю JekyllЖ│╝ LiquidвАю В╣┤ьЁїЖ│авдг ВІюВіцьЁюВЮў HTML ЖхгВА░вЦ╝ вДївЊцВЌѕВіхвІѕвІц. ВЮ┤в▓ѕВЌљвіћ JavaScriptвЦ╝ ВѓгВџЕьЋ┤Вёю ВѓгВџЕВъљЖ░ђ ВІцВаювАю ьЂ┤вдГьЋўЖ│а ВЃЂьўИВъЉВџЕьЋа Вѕў Въѕвіћ вЈЎВаЂ ЖИ░віЦВЮё ЖхгьўёьЋ┤в│┤Ж▓аВіхвІѕвІц.

ве╝Вађ DOM(Document Object Model)ВЌљ вїђьЋ┤ Ж░ёвІеьъѕ ВЋїВЋёв│┤Ж▓аВіхвІѕвІц. DOMВЮђ ВЏ╣ьјўВЮ┤ВДђВЮў HTML ЖхгВА░вЦ╝ JavaScriptВЌљВёю ВА░ВъЉьЋа Вѕў ВъѕЖ▓ї ьЋ┤ВБ╝віћ ВЮИьё░ьјўВЮ┤ВіцВъЁвІѕвІц.

ВЮ┤в▓цьіИ вдгВіцвёѕ вЊ▒вАЮьЋўЖИ░

В╣┤ьЁїЖ│авдг ВІюВіцьЁюВЌљ ьЋёВџћьЋю ВЮ┤в▓цьіИ вдгВіцвёѕвЊцВЮё вЊ▒вАЮьЋ┤в│┤Ж▓аВіхвІѕвІц:

// categories.html ьЋўвІеВЌљ ВХћЖ░ђ
<script>
document.addEventListener('DOMContentLoaded', function() {
    // ьЋёВџћьЋю DOM ВџћВєївЊц ВёаьЃЮ
    const categoryItems = document.querySelectorAll('.category-item');
    const postsContainers = document.querySelectorAll('.posts-container');
    const categoryContent = document.querySelector('.category-content');

    // В╣┤ьЁїЖ│авдг ьЂ┤вдГ ВЮ┤в▓цьіИ В▓ўвдг
    categoryItems.forEach(item => {
        item.addEventListener('click', function() {
            const category = this.dataset.category;

            // вфевЊа В╣┤ьЁїЖ│авдгВЌљВёю active ьЂ┤въўВіц ВаюЖ▒░
            categoryItems.forEach(cat => cat.classList.remove('active'));

            // ьЂ┤вдГвљю В╣┤ьЁїЖ│авдгВЌљ active ьЂ┤въўВіц ВХћЖ░ђ
            this.classList.add('active');

            // ьЋ┤вІ╣ В╣┤ьЁїЖ│авдгВЮў В╗еьЁљВИа ьЉюВІю
            showCategoryContent(category);
        });
    });

    // ВёювИїВ╣┤ьЁїЖ│авдг ьЂ┤вдГ ВЮ┤в▓цьіИ В▓ўвдг
    function setupSubcategoryEvents() {
        const subcategoryCards = document.querySelectorAll('.subcategory-card');

        subcategoryCards.forEach(card => {
            card.addEventListener('click', function() {
                const subcategory = this.dataset.subcategory;
                showPostsBySubcategory(subcategory);
            });
        });
    }

    // В╣┤ьЁїЖ│авдг В╗еьЁљВИа ьЉюВІю ьЋеВѕў
    function showCategoryContent(category) {
        postsContainers.forEach(container => {
            if (container.dataset.category === category) {
                container.style.display = 'block';

                // ВЋавІѕвЕћВЮ┤ВЁў ьџеЖ│╝
                container.style.opacity = '0';
                setTimeout(() => {
                    container.style.opacity = '1';
                }, 50);
            } else {
                container.style.display = 'none';
            }
        });

        // ВёювИїВ╣┤ьЁїЖ│авдг ВЮ┤в▓цьіИ ВъгВёцВаЋ
        setupSubcategoryEvents();
    }

    // ВёювИїВ╣┤ьЁїЖ│авдгв│ё ьЈгВіцьіИ ьЉюВІю ьЋеВѕў
    function showPostsBySubcategory(subcategory) {
        const allPosts = document.querySelectorAll('.post-item');

        allPosts.forEach(post => {
            if (post.dataset.subcategory === subcategory) {
                post.style.display = 'block';
                post.classList.add('fade-in');
            } else {
                post.style.display = 'none';
                post.classList.remove('fade-in');
            }
        });
    }

    // В┤ѕЖИ░ вАювЊю ВІю В▓Ф в▓ѕВДИ В╣┤ьЁїЖ│авдг ьЎюВё▒ьЎћ
    if (categoryItems.length > 0) {
        categoryItems[0].click();
    }
});
</script>

ВЮ┤в▓цьіИ вдгВіцвёѕвъђ? ВЏ╣ьјўВЮ┤ВДђВЌљВёю ВѓгВџЕВъљВЮў вЈЎВъЉ(ВЮ┤в▓цьіИ)ВЮё Ж░љВДђьЋўЖ│а в░ўВЮЉьЋўвіћ Въљв░ћВіцьЂгвдйьіИ ьЋеВѕўВъЁвІѕвІц.

ВБ╝Вџћ ВЮ┤в▓цьіИ ВбЁвЦўвАювіћ

  • click: вДѕВџ░Віц ьЂ┤вдГ
  • input: ВъЁваЦ ьЋёвЊюВЌљ ьЁЇВіцьіИ ВъЁваЦ
  • keydown: ьѓцв│┤вЊю вѕёвдё
  • scroll: ВіцьЂгвАц
  • load: ьјўВЮ┤ВДђ вАювћЕ ВЎёвБї

ВЮ┤в▓цьіИЖ░ђ ВЮ╝Вќ┤вѓўвЕ┤ Вќ┤вќц вЈЎВъЉВЮё ьЋаВДђ ВЮ┤ваЄЖ▓ї ьЋўвЕ┤ ВёцВаЋьЋа Вѕў ВъѕВіхвІѕвІц. вг╝вАа вДцВџ░ В┤ѕв│┤ВаЂВЮИ вѓ┤ВџЕВЮ┤ЖИ░ вЋївгИВЌљ ВЏљьЋўвіћ ЖИ░віЦВЮ┤ ВъѕвІцвЕ┤ вЇћ Ж│хвХђьЋ┤ВЋ╝Ж▓аВБа! ЖиИВ╣ўвДї ВЮ┤ вИћвАюЖиИВЌљВёю ВаюЖ░ђ ьЋўЖ│аВІХВЮђ вЈЎВаЂ В╣┤ьЁїЖ│авдг ВІюВіцьЁюВЮђ ВЮ┤ ВаЋвЈё ВъЁвІѕвІц.
