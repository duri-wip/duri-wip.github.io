def solution(s):
    # ьїїВІ▒
    ## вгИВъљВЌ┤ВЌљВёю ВДЉьЋЕьўЋьЃю ВаюЖ▒░ьЋўЖ│а вХёвдгьЋўЖИ░
    s = s[2:-2].replace('},{','-')
    s = s.split('-')

    ## ВѕФВъљьўЋВю╝вАю в│ђьЎўьЋўЖ│а ВЁІВю╝вАю вДївЊцЖИ░
    new_s = [set(map(int, x.split(','))) for x in s]

    ## ЖИИВЮ┤ВЌљ вћ░вЮ╝ ВўцвдёВ░еВѕю ВаЋвагьЋўЖИ░
    new_s.sort(key=len)

    # ВЃѕвАюВЮ┤ ВХћЖ░ђвљю ВЏљВєї В░ЙВЋёВёю в░ўьЎўьЋўЖИ░
    result = []
    for subset in new_s:
        number = (list(subset-set(result))[0])
        result.append(number)

    return result
