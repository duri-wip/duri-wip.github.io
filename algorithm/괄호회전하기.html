def isValid(s):
    tmp = []
    for i in s:
        if i in ["(","[","{"]:
            tmp.append(i)
        else:
            if i==")":
                if not tmp:#м•„л¬ҙкІғлҸ„ м—ҶлҠ”лҚ° лҒқкҙ„нҳёл§Ң л“Өм–ҙк°Җл©ҙ f

                elif tmp[-1]=="(":#л§Ҳм§Җл§ү мҡ”мҶҢк°Җ мІЁкҙ„нҳёл©ҙ нҢқ

                    return False
            elif i == "]":
                if not tmp:
                    return False

                    tmp.pop()
                else:
                    return False
            else:
                if not tmp:
                    return False
                elif tmp[-1]=="{":
                    tmp.pop()
                else:
                    return False
    return not tmp

def toleft(s):
    return(s[1:]+s[0])

def solution(s):
    res = 0
    for i in range(len(s)-1):
        #print(isValid(s))
        res += isValid(s)
        s = toleft(s)
        #print(s)
    return res
