class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        if len(s) == len(t):
            s = [c for c in s]
            t = [c for c in t]

            while s:
                if s[-1] in t:
                    t.pop(t.index(s[-1]))
                    s.pop()
                else:
                    return False
            return True
        else:
            return False
