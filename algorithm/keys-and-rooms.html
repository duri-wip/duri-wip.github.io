class Solution:
    def canVisitAllRooms(self, rooms: List[List[int]]) -> bool:
        visited = [False]*len(rooms)

        def dfs(cur_v, rooms):
            visited[cur_v]=True
            for next_v in rooms[cur_v]:
                if not visited[next_v]:
                    dfs(next_v,rooms)
        
        dfs(0, rooms)       
        if sum(visited) < len(rooms):
            return False
        else: return True
