def solution(n, works):
    if sum(works) <= n:
        return 0

    while n > 0:
        idx = works.index(max(works))
        works[idx] -= 1
        n -= 1
    
    def cal_p(n):
        return n ** 2
    
    return sum(map(cal_p, works))

import heapq

def solution(n, works):
    if sum(works) <= n:
        return 0

    # Вхювїђ ьъЎ ВЃЮВё▒ (ВЮїВѕў ВѓгВџЕ)
    works = [-work for work in works]
    heapq.heapify(works)

    while n > 0:
        max_work = heapq.heappop(works)
        if max_work == 0:
            break  # ВъЉВЌЁвЪЅВЮ┤ 0ВЮИ Ж▓йВџ░ ВцЉвІе
        heapq.heappush(works, max_work + 1)  # ВъЉВЌЁвЪЅ Ж░љВєї
        n -= 1

    # ВЋ╝Жи╝ ВДђВѕў Ж│ёВѓ░
    return sum([work ** 2 for work in works])
