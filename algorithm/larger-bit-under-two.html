def solution(numbers):
    result = []

    for num in numbers:
        if num % 2 == 0:
            # ВДЮВѕўВЮў Ж▓йВџ░ вг┤ВА░Ж▒┤ ВўцвЦИВфй вДѕВДђвДЅ ВѕФВъљЖ░ђ 0ВЮ┤вІц.
            result.append(num + 1)
        else:
            # ьЎђВѕўВЮў Ж▓йВџ░ Ж░ђВъЦ вДѕВДђвДЅВЌљ Въѕвіћ 0ВъљвдгВѕўвЦ╝ В░ЙВЋёВЋ╝ ьЋювІц.
            bnum = bin(num)[2:]
            n = len(bnum)
            idx = bnum.rfind('0')
            result.append(num + (1 << (n - idx - 1)) - (1 << (n - idx - 2)))
    return result
