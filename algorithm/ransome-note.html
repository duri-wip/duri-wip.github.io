문제

magazine에 있는 문자로 ransomnote를 쓸 수 있으면 t, 아니면 f

풀이

class Solution:
    def canConstruct(self, ransomNote: str, magazine: str) -> bool:
        r = [w for w in ransomNote]
        m = [w for w in magazine]

        for w in r:
            if w in m:
                m.pop(m.index(w))
            else:
                return False
        return True
