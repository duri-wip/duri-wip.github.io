class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        # ВаЋвагвљю в░░ВЌ┤ВЌљВёю Вќ┤вќцЖ▓ї ьЃђЖ▓ЪВЮИВДђ В░ЙВЋёВўцЖИ░
        start = 0
        fin  = len(nums) -1
        # в░░ВЌ┤ВЮ┤ в╣ёВќ┤Въѕвіћ Ж▓йВџ░
        if not nums :
            return [-1, -1]
        # ьЃђЖ▓ЪВЮ┤ в░░ВЌ┤ ВЋѕВЌљ ВЌєвіћ Ж▓йВџ░
        elif nums[start] > target or nums[fin] < target:
            return [-1, -1]
        # ьЃђЖ▓ЪВЮ┤ ВъѕВЮёВДђвЈё вфевЦ┤віћ Ж▓йВџ░
        else:
            result = []
            for i in range(len(nums)):
                if nums[i] == target:
                    result.append(i)
            if not result:
                return [-1, -1]
            else:
                return [min(result), max(result)]
