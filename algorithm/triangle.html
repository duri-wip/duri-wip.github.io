def solution(n):
    #вфЕьЉю
    target = sum(range(1, n+1))
    
    # Вѓ╝Ж░ЂьўЋ В┤ѕЖИ░ьЎћ
    trian = [[0]*i for i in range(1,n+1)]

    #┬авІцвЦИ в│ђВѕў В┤ѕЖИ░ьЎћ
    num = 1
    x,y = 0,0
    d = 0

    # в░ЕьќЦ
    dirs = [(1,0),(0,1),(-1,-1)]

    #ВЮ┤вЈЎ
    while num <= target:
        trian[x][y] = num
        num += 1

        # вІцВЮїв░ЕьќЦВю╝вАю ВЮ┤вЈЎ
        nx, ny = x + dirs[d][0], y + dirs[d][1]

        # вІцВЮї ВюёВ╣ўЖ░ђ ВюаьџеьЋюВДђ ьЎЋВЮИ
        if nx < 0 or ny <0 or nx >= n or ny >= len(trian[nx]) or trian[nx][ny] != 0:
            d = (d + 1) % 3
            nx, ny = x + dirs[d][0], y + dirs[d][1]
        
        x, y = nx, ny
    
    result = []
    for row in trian:
        result.extend(row)
    return result
